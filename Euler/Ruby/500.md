def prime_factors(n)
  factors = Hash.new(0)
  d = 2
  
  while d * d <= n
    while n % d == 0
      factors[d] += 1
      n /= d
    end
    d += 1
  end
  
  factors[n] += 1 if n > 1
  factors
end
