# Euler Problem 174 Solution in Visual Basic

This problem asks for the number of values of n ≤ 1000000 for which the number of ways to tile a 1×n rectangle using 1×1 and 1×2 tiles is exactly 10.

```vb
Module Problem174
    Sub Main()
        Dim limit As Long = 1000000
        Dim count(1000000) As Integer
        Dim result As Long = 0
        
        ' Initialize count array
        For i As Integer = 0 To 1000000
            count(i) = 0
        Next
        
        ' For each possible rectangle size
        For n As Integer = 1 To limit
            ' Count the number of ways to tile a 1×n rectangle
            Dim ways As Long = CountWays(n)
            If ways <= 1000000 Then
                count(ways) += 1
            End If
        Next
        
        ' Count how many numbers have exactly 10 representations
        For i As Integer = 1 To 1000000
            If count(i) = 10 Then
                result += 1
            End If
        Next
        
        Console.WriteLine("Answer: " & result)
        Console.ReadKey()
    End Sub
    
    Function CountWays(n As Integer) As Long
        If n = 0 Then Return 1
        If n = 1 Then Return 1
        If n = 2 Then Return 2
        
        Dim dp(n) As Long
        dp(0) = 1
        dp(1) = 1
        dp(2) = 2
        
        For i As Integer = 3 To n
            dp(i) = dp(i - 1) + dp(i - 2)
        Next
        
        Return dp(n)
    End Function
End Module
```

Module Problem174
    Sub Main()
        Dim limit As Long = 1000000
        Dim ways(1000000) As Integer
        Dim result As Long = 0
        
        ' Initialize ways array
        For i As Integer = 0 To 1000000
            ways(i) = 0
        Next
        
        ' For each possible hollow square lamina
        ' A hollow square lamina with outer dimensions (a×a) and inner dimensions (b×b)
        ' has area a² - b² where 0 < b < a
        For a As Integer = 2 To CInt(Math.Sqrt(limit)) + 1
            For b As Integer = 1 To a - 1
                Dim area As Long = a * a - b * b
                If area <= limit Then
                    ways(area) += 1
                End If
            Next
        Next
        
        ' Count how many values have exactly 10 ways
        For i As Integer = 1 To limit
            If ways(i) = 10 Then
                result += 1
            End If
        Next
        
        Console.WriteLine("Answer: " & result)
        Console.ReadKey()
    End Sub
End Module
