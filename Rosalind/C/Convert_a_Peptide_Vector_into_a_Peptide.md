char* peptide_vector_to_peptide(int* vector, int length) {

    // Calculate the total mass from the vector
    int total_mass = 0;
    for (int i = 0; i < length; i++) {
        if (vector[i] == 1) {
            total_mass += (i + 1);
        }
    }
    
    // Create result string
    int peptide_length = total_mass / 14;
    char* peptide = (char*)malloc((peptide_length + 1) * sizeof(char));
    peptide[peptide_length] = '\0';
    
    // Reconstruct peptide
    int current_mass = 0;
    int peptide_index = 0;
    
    for (int i = 0; i < length; i++) {

                if (masses[j] == (i + 1)) {
                    peptide[peptide_index++] = letters[j];
                    break;
                }
            }
        }
    }
    
    return peptide;
}
